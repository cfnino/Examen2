cons = []
for b in range(0, len(blocks)):
    for r in range(1,len(blocks[b])):
        content = blocks[b][r].find_all('td')
        con = []
        for c in range(0, len(content)):
            if(c != 0):
                # Como estamos utilizando los ttulos de la primera tabla, significa que nos falta
                # una columna por rellenar para las siguientes tablas, por esta razn validamos que estemos 
                # en una tabla diferente a la primera y agregamos un campo vaco antes de la columna 4 que hara referencia al ttulo de ciudades.
                if(c == 4 and b!=0):
                    con.append('')
                    con.append(content[c].text.strip())
                else:
                    # Para la columna dos est la imagen del presidente por esta razn volvemos a buscar un tag
                    # que hace referencia a la imagen, el tag que buscamos es img y obtenemos el valor src de este tag
                    if(c == 2):
                        img = content[c].find_all('img')
                        # Validamos si tiene informacin
                        if(len(img) > 0):
                            con.append(img[0]['src'])
                        else:
                            con.append('')
                    else:
                        con.append(content[c].text.strip())
        cons.append(con)
